-- Clean Souls Hub UI Test Script
-- This is a minimal script to test only the UI framework

-- Load the library with error handling
local success, SoulsHub = pcall(function()
    return loadstring(game:HttpGet("https://pandadevelopment.net/virtual/file/e7f388d3c065df7a"))()
end)

if not success or not SoulsHub then
    warn("Failed to load SoulsHub library")
    return
end

-- Wait for library initialization (critical)
task.wait(1)

-- Call the loader (as per documentation)
if SoulsHub.Loader then
    pcall(function() SoulsHub:Loader(nil, 1).yield() end)
end

-- Create the window (simplified)
local Window
pcall(function()
    Window = SoulsHub.new({
        Keybind = "LeftAlt"
    })
end)

if not Window then
    warn("Failed to create window")
    return
end

-- Set up basic UI
Window:Watermark()
Window.Username = game.Players.LocalPlayer.Name

-- Add a simple tab and section
local MainTab = Window:DrawTab({
    Icon = "info",
    Name = "Test",
    Type = "Single"
})

local section = MainTab:DrawSection({
    Name = "Basic Test",
    Position = "LEFT"
})

section:AddParagraph({
    Title = "UI Test",
    Content = "If you see this, the UI framework is working correctly."
})

-- Draw the UI (must be called at the end)
Window:Draw()
